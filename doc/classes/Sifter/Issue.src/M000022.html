    def update(updates)
      updates.reject! {|key, value| value.nil? || value.strip == '' }
      
      if updates.empty?
        return Sifter.detailed_return(project.client.detailed_return,
          :successful => true,
          :message => 'No updates or changes made.')
      end
      
      update_status_and_body(updates)
      update_others(updates)
    end
